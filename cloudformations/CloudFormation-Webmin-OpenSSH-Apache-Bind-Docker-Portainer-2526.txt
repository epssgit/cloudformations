AWSTemplateFormatVersion: '2010-09-09'
Description: Deploys an Ubuntu 22.04 instance with Webmin, OpenSSH, Apache, Bind with Elastic IP

Parameters:

  1VPC:
    Description: VPC where the instance will be deployed
    Type: AWS::EC2::VPC::Id

  2Subnet:
    Description: Subnet for the instance
    Type: AWS::EC2::Subnet::Id  
    
  3InstanceName:
    Type: String
    Description: EC2 Instance name
  
  4InstanceType:
    Description: EC2 instance type
    Type: String
    Default: t2.micro
    AllowedValues:
      - t2.micro
      - t3.small
      - t3.medium

  5KeyName:
    Description: SSH key pair to access the EC2 instance
    Type: AWS::EC2::KeyPair::KeyName

Resources:
  InstanceSecurityGroup:
    Type: AWS::EC2::SecurityGroup
    Properties:
      GroupDescription: Allow Webmin, OpenSSH, Apache
      VpcId: !Ref 1VPC
      SecurityGroupIngress:

        # Allow SSH access
        - IpProtocol: tcp
          FromPort: 22
          ToPort: 22
          CidrIp: 0.0.0.0/0

        # Allow Webmin access
        - IpProtocol: tcp
          FromPort: 10000
          ToPort: 10000
          CidrIp: 0.0.0.0/0

        # Allow Apache access
        - IpProtocol: tcp
          FromPort: 80
          ToPort: 80
          CidrIp: 0.0.0.0/0
 
        # Allow DNS access
        - IpProtocol: tcp
          FromPort: 53
          ToPort: 53
          CidrIp: 0.0.0.0/0

        - IpProtocol: udp
          FromPort: 53
          ToPort: 53
          CidrIp: 0.0.0.0/0

        # Allow Web Admin Portainer
        - IpProtocol: tcp
          FromPort: 9443
          ToPort: 9443
          CidrIp: 0.0.0.0/0


  EC2Instance:
    Type: AWS::EC2::Instance
    Properties:
      InstanceType: !Ref 4InstanceType
      KeyName: !Ref 5KeyName
      ImageId: ami-0e1bed4f06a3b463d  # AMI Ubuntu 22.04
      SubnetId: !Ref 2Subnet
      SecurityGroupIds:
        - !Ref InstanceSecurityGroup
      Tags:
        - Key: Name
          Value: !Ref 3InstanceName
      UserData:
        Fn::Base64: !Sub |
          #!/bin/bash
          # Update and install basic dependencies
          apt update && apt upgrade -y
          
          # Install OpenSSH Server
          apt install -y openssh-server
          systemctl enable ssh
          systemctl start ssh

          # Install Apache
          apt install apache2 -y
          systemctl start apache2
          systemctl enable apache2

          # Install BIND
          apt install -y bind9 bind9utils bind9-doc
          systemctl enable bind9
          systemctl start bind9

          # Install Docker
          curl -fsSL https://download.docker.com/linux/ubuntu/gpg | gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
          echo "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu jammy stable" | tee /etc/apt/sources.list.d/docker.list
          apt update && apt install -y docker-ce docker-ce-cli containerd.io

          # Add the ubuntu user to the Docker group
          usermod -aG docker ubuntu

          # Enable and start Docker service
          systemctl enable --now docker
       
          # Deploy Portainer
          docker volume create portainer_data
          docker run -d -p 8000:8000 -p 9443:9443 --name portainer --restart=always -v /var/run/docker.sock:/var/run/docker.sock -v portainer_data:/data portainer/portainer-ce:lts

          # Webmin install
          wget -q -O- http://www.webmin.com/jcameron-key.asc | gpg --dearmor -o /usr/share/keyrings/webmin.gpg
          echo "deb [signed-by=/usr/share/keyrings/webmin.gpg] http://download.webmin.com/download/repository sarge contrib" | tee /etc/apt/sources.list.d/webmin.list
          apt update && apt install -y webmin

          # Configure the admin user
          /usr/share/webmin/changepass.pl /etc/webmin root 12345aA
          systemctl restart webmin
 
  ElasticIP:
    Type: AWS::EC2::EIP
    Properties:
      InstanceId: !Ref EC2Instance
